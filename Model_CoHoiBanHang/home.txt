    <?php
    // Thông tin kết nối cơ sở dữ liệu
    $servername = 'localhost';
    $username = 'root';
    $password = '';
    $dbname = 'db_crm';

    // Tạo kết nối
    $conn = new mysqli($servername, $username, $password, $dbname);

    // Kiểm tra kết nối
    if ($conn->connect_error) {
        die("Kết nối thất bại: " . $conn->connect_error);
    }

    // Lấy danh sách nhân viên để điền vào combobox `NguoiPhuTrach`
    $nvQuery = "SELECT HovaTen FROM nhanvien";
    $nvResult = $conn->query($nvQuery);

    // Lấy danh sách các giá trị cho các combobox khác
    $tinhTrangQuery = "SELECT DISTINCT TinhTrang FROM khachhangch";
    $nguonCoHoiQuery = "SELECT DISTINCT NguonCoHoi FROM khachhangch";
    $khuVucQuery = "SELECT DISTINCT KhuVuc FROM khachhangch";

    $tinhTrangResult = $conn->query($tinhTrangQuery);
    $nguonCoHoiResult = $conn->query($nguonCoHoiQuery);
    $khuVucResult = $conn->query($khuVucQuery);

    // Truy vấn dữ liệu từ bảng `khachhangch`
    $sql = "SELECT Id_KHCH, Ten, TenCongTy, Email, Phone, NguoiPhuTrach, TinhTrang, NguonCoHoi, NgayLienHe, Website, KhuVuc, GiaDuKien, NgayChotDuKien, MoTa FROM khachhangch";
    $result = $conn->query($sql);

    // Kiểm tra và hiển thị dữ liệu
    if ($result->num_rows > 0) {
        echo "<form method='post' action=''>";
        echo "<table>";
        echo "<tr>
            <th>STT</th>
            <th>Ten</th>
            <th>TenCongTy</th>
            <th>Email</th>
            <th>Phone</th>
            <th>NguoiPhuTrach</th>
            <th>TinhTrang</th>
            <th>NguonCoHoi</th>
            <th>NgayLienHe</th>
            <th>KhuVuc</th>
            <th>GiaDuKien</th>
        </tr>";
        
        // Lưu dữ liệu combobox vào biến
        $nguoiPhuTrachOptions = '';
        while ($nvRow = $nvResult->fetch_assoc()) {
            $nguoiPhuTrachOptions .= "<option value='" . $nvRow["HovaTen"] . "'>" . $nvRow["HovaTen"] . "</option>";
        }

        $tinhTrangOptions = '';
        while ($tinhTrangRow = $tinhTrangResult->fetch_assoc()) {
            $tinhTrangOptions .= "<option value='" . $tinhTrangRow["TinhTrang"] . "'>" . $tinhTrangRow["TinhTrang"] . "</option>";
        }
        $tinhTrangOptions .= "<option value='Mới'>Mới</option>";
        $tinhTrangOptions .= "<option value='Đang xử lý'>Đang xử lý</option>";

        $nguonCoHoiOptions = '';
        while ($nguonCoHoiRow = $nguonCoHoiResult->fetch_assoc()) {
            $nguonCoHoiOptions .= "<option value='" . $nguonCoHoiRow["NguonCoHoi"] . "'>" . $nguonCoHoiRow["NguonCoHoi"] . "</option>";
        }
        $nguonCoHoiOptions .= "<option value='Sự kiện'>Sự kiện</option>";
        $nguonCoHoiOptions .= "<option value='Hội thảo'>Hội thảo</option>";

        $khuVucOptions = '';
        while ($khuVucRow = $khuVucResult->fetch_assoc()) {
            $khuVucOptions .= "<option value='" . $khuVucRow["KhuVuc"] . "'>" . $khuVucRow["KhuVuc"] . "</option>";
        }
        $khuVucOptions .= "<option value='Khu vực mới'>Khu vực mới</option>";
        $khuVucOptions .= "<option value='Khu vực khác'>Khu vực khác</option>";

        // Lặp qua từng dòng kết quả
        $stt = 1; // Khởi tạo biến STT
        while ($row = $result->fetch_assoc()) {
            echo "<tr>";
            echo "<td>" . $stt++ . "</td>";
            echo "<td>" . $row["Ten"] . "</td>";
            echo "<td>" . $row["TenCongTy"] . "</td>";
            echo "<td>" . $row["Email"] . "</td>";
            echo "<td>" . $row["Phone"] . "</td>";

            // Hiển thị combobox cho NguoiPhuTrach
            echo "<td><select name='nguoiPhuTrach[]'>";
            echo "<option value=''>Chọn NguoiPhuTrach</option>"; // Thêm tùy chọn trống
            echo $nguoiPhuTrachOptions;
            echo "</select></td>";

            // Hiển thị combobox cho TinhTrang
            echo "<td><select name='tinhTrang[]'>";
            echo "<option value=''>Chọn TinhTrang</option>"; // Thêm tùy chọn trống
            echo $tinhTrangOptions;
            echo "</select></td>";

            // Hiển thị combobox cho NguonCoHoi
            echo "<td><select name='nguonCoHoi[]'>";
            echo "<option value=''>Chọn NguonCoHoi</option>"; // Thêm tùy chọn trống
            echo $nguonCoHoiOptions;
            echo "</select></td>";

            // Hiển thị combobox cho KhuVuc
            echo "<td><select name='khuVuc[]'>";
            echo "<option value=''>Chọn KhuVuc</option>"; // Thêm tùy chọn trống
            echo $khuVucOptions;
            echo "</select></td>";

            // Hiển thị NgayLienHe
            echo "<td>" . $row["NgayLienHe"] . "</td>";
            echo "<td>" . $row["GiaDuKien"] . "</td>";
            echo "</tr>";
        }
        echo "</table>";
        echo "<input type='submit' value='Lưu Thay Đổi'>";
        echo "</form>";
    } else {
        echo "Không có dữ liệu nào.";
    }

    // Đóng kết nối
    $conn->close();
    ?>
